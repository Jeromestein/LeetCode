    result = 10 / 0

    print("Cannot divide by zero!")

class CustomError(Exception):
    pass

def validate_age(age):
    if age < 0:
        raise CustomError("Age cannot be negative")
    return age

# Usage

    validate_age(-5)
except CustomError as e:
    print(f"Validation error: {e}")

Function Definition

üèóÔ∏è Object-Oriented Programming

Class Definition

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def greet(self):
        return f"Hello, I'm {self.name}"
    
    @property
    def is_adult(self):
        return self.age >= 18
    
    def __str__(self):
        return f"Person({self.name}, {self.age})"

# Usage
person = Person("Alice", 25)
print(person.greet())
print(person.is_adult)

Inheritance

class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        pass

class Dog(Animal):
    def speak(self):
        return f"{self.name} says Woof!"

class Cat(Animal):
    def speak(self):
        return f"{self.name} says Meow!"

# Usage
dog = Dog("Buddy")
cat = Cat("Whiskers")
print(dog.speak())
print(cat.speak())

Collections

Generators
# Generator function
def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

# Generator expression
squares = (x**2 for x in range(5))

# Usage
for num in fibonacci(10):
    print(num)
Context Managers
# Custom context manager
class Timer:
    def __init__(self, name):
        self.name = name
    
    def __enter__(self):
        import time
        self.start = time.time()
        return self
    
    def __exit__(self, *args):
        import time
        print(f"{self.name} took {time.time() - self.start:.2f}s")

# Usage
with Timer("my_function"):
    # do something
    pass
Regular Expressions
import re

# Basic patterns

if match:
    print(match.group())  # '123'

# Replace
new_text = re.sub(pattern, 'NUMBER', text)
