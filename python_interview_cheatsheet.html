Variables and Assignment

# Variable assignment
name = "Python"
age = 25
is_student = True
pi = 3.14159

Data Types

text = "Hello"          # str
number = 42             # int
decimal = 3.14          # float
boolean = True          # bool
empty = None            # NoneType

# Type checking
print(type(text))       # <class 'str'>
print(isinstance(number, int))  # True

String Operations

# String methods
text = "  Hello World  "
print(text.strip())           # "Hello World"
print(text.upper())           # "  HELLO WORLD  "
print(text.lower())           # "  hello world  "
print(text.replace("World", "Python"))

# String formatting
name = "Alice"
age = 25
print(f"{name} is {age} years old")

üìä Data Structures

List

# Create list
numbers = [1, 2, 3, 4, 5]
mixed = [1, "hello", 3.14, True]

# Common operations
numbers.append(6)       # Add element
numbers.insert(0, 0)    # Insert element
numbers.remove(3)       # Remove element
popped = numbers.pop()  # Pop last element

Dictionary

# Create dictionary
person = {
    "name": "Alice",
    "age": 25,
    "city": "Beijing"
}

# Common operations
person["email"] = "alice@example.com"  # Add key-value pair
age = person.get("age", 0)             # Safe get
del person["city"]                     # Delete key-value pair

# Iteration
for key, value in person.items():
    print(f"{key}: {value}")

# Dictionary comprehension
squares = {x: x**2 for x in range(5)}

Set

# Create set
numbers = {1, 2, 3, 4, 5}
fruits = set(["apple", "banana", "apple"])

# Set operations
numbers.add(6)          # Add element
numbers.remove(1)       # Remove element
numbers.discard(10)     # Safe remove

# Set operations
set1 = {1, 2, 3}
set2 = {3, 4, 5}
union = set1 | set2     # Union
intersection = set1 & set2  # Intersection
difference = set1 - set2    # Difference

Tuple

# Create tuple

person = ("Alice", 25, "Beijing")

# Tuples are immutable

# Unpacking

name, age, city = person

# Named tuple
from collections import namedtuple
Point = namedtuple('Point', ['x', 'y'])
p = Point(10, 20)
print(p.x, p.y)
üîÑ Control Flow

Conditional Statements

# if-elif-else
age = 18
if age < 13:
    print("Child")
elif age < 20:
    print("Teenager")
else:
    print("Adult")

# Ternary operator
status = "adult" if age >= 18 else "minor"

# Conditional expression
result = "positive" if x > 0 else "negative" if x < 0 else "zero"

Loops

# for loop
for i in range(5):
    print(i)

for item in ["apple", "banana", "cherry"]:
    print(item)

# while loop
count = 0
while count < 5:
    print(count)
    count += 1

# Loop control
for i in range(10):
    if i == 5:
        break      # Exit loop
    if i == 2:
        continue   # Skip current iteration

List Comprehension

# Basic list comprehension
squares = [x**2 for x in range(5)]
# Result: [0, 1, 4, 9, 16]

# List comprehension with condition
even_squares = [x**2 for x in range(10) if x % 2 == 0]
# Result: [0, 4, 16, 36, 64]

# Nested list comprehension
matrix = [[i+j for j in range(3)] for i in range(3)]
# Result: [[0, 1, 2], [1, 2, 3], [2, 3, 4]]
‚öôÔ∏è Functions

Function Definition

# Basic function
def greet(name):
    return f"Hello, {name}!"

# Default parameters
def greet_with_default(name="World"):
    return f"Hello, {name}!"

# Variable arguments
def sum_numbers(*args):
    return sum(args)

# Keyword arguments
def person_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

# Function calls
print(greet("Alice"))
print(sum_numbers(1, 2, 3, 4, 5))
person_info(name="Alice", age=25, city="Beijing")

Lambda Functions

# Lambda expressions
square = lambda x: x**2
add = lambda x, y: x + y

# Using with functions
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, numbers))
filtered = list(filter(lambda x: x > 2, numbers))

# Sorting
students = [("Alice", 25), ("Bob", 20), ("Charlie", 22)]
students.sort(key=lambda x: x[1])  # Sort by age

Decorators

# Basic decorator
def timer(func):
    def wrapper(*args, **kwargs):
        import time
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} took {end-start:.2f} seconds")
        return result
    return wrapper

@timer
def slow_function():
    import time
    time.sleep(1)
    return "Done!"

# Usage
slow_function()
üèóÔ∏è Object-Oriented Programming

Class Definition

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def greet(self):
        return f"Hello, I'm {self.name}"
    
    @property
    def is_adult(self):
        return self.age >= 18
    
    def __str__(self):
        return f"Person({self.name}, {self.age})"

# Create instance
person = Person("Alice", 25)
print(person.greet())
print(person.is_adult)

Inheritance

class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        pass

class Dog(Animal):
    def speak(self):
        return f"{self.name} says Woof!"

class Cat(Animal):
    def speak(self):
        return f"{self.name} says Meow!"

# Usage
dog = Dog("Buddy")
cat = Cat("Whiskers")
print(dog.speak())
print(cat.speak())
‚ö†Ô∏è Exception Handling
# Basic exception handling
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")
except Exception as e:
    print(f"An error occurred: {e}")
else:
    print("No exception occurred")
finally:
    print("This always executes")

# Custom exceptions
class CustomError(Exception):
    pass

def validate_age(age):
    if age < 0:
        raise CustomError("Age cannot be negative")
    return age

# Usage
try:
    validate_age(-5)
except CustomError as e:
    print(f"Validation error: {e}")
üì¶ Common Modules

Collections

from collections import Counter, defaultdict, deque

# Counter - counting
words = ["apple", "banana", "apple", "cherry"]
word_count = Counter(words)
print(word_count)  # Counter({'apple': 2, 'banana': 1, 'cherry': 1})

# defaultdict - default dictionary
dd = defaultdict(list)
dd["fruits"].append("apple")
dd["fruits"].append("banana")

# deque - double-ended queue
d = deque([1, 2, 3])
d.append(4)
d.appendleft(0)
print(d)  # deque([0, 1, 2, 3, 4])

itertools

from itertools import combinations, permutations, product

# Combinations
items = [1, 2, 3]
combs = list(combinations(items, 2))
# [(1, 2), (1, 3), (2, 3)]

# Permutations
perms = list(permutations(items, 2))
# [(1, 2), (1, 3), (2, 1), (2, 3), (3, 1), (3, 2)]

# Cartesian product
prod = list(product([1, 2], ['a', 'b']))
# [(1, 'a'), (1, 'b'), (2, 'a'), (2, 'b')]
